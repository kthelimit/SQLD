CREATE TABLE ORDERS(
    ORDER_ID NUMBER(12,0) PRIMARY KEY,
    ORDER_DATE DATE,
    ORDER_MODE VARCHAR2(8 byte) CONSTRAINTS MODECHECK CHECK(ORDER_MODE IN('direct', 'online')),
    CUSTOMER_ID NUMBER(6,0),
    ORDER_STATUS NUMBER(2,0),
    ORDER_TOTAL NUMBER(8,2) DEFAULT 0,
    SALES_REP_ID NUMBER(6,0),
    PROMOTION_ID NUMBER(6,0)
);

CREATE TABLE ORDER_ITEMS(
    ORDER_ID NUMBER(12,0),
    LINE_ITEM_ID NUMBER(3,0),
    PRODUCT_ID NUMBER(3,0),
    UNIT_PRICE NUMBER(8,2) DEFAULT 0,
    QUANTITY NUMBER(8,0) DEFAULT 0,
    CONSTRAINT PK_ORDER PRIMARY KEY(ORDER_ID, LINE_ITEM_ID)
);

CREATE TABLE PROMOTIONS(
    PROMO_ID NUMBER(6,0),
    PROMO_NAME VARCHAR2(20),
    CONSTRAINT PK_PROMO PRIMARY KEY(PROMO_ID)
);

CREATE SEQUENCE ORDERS_SEQ
INCREMENT BY 1
START WITH 1
MINVALUE 1
MAXVALUE 9999
NOCYCLE
NOCACHE;
